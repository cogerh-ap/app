Tutorial e Código Google Apps Script

Para fazer a sua Planilha do Google Sheets disparar notificações push diretamente para os usuários do PWA quando você adicionar/editar avisos ou clicar em um botão, siga estes passos:

Passo 1: Abrir o Editor de Scripts

Na sua Planilha do Google, clique no menu superior em Extensões -> Apps Script.

Passo 2: Colar o Código do Script

Apague todo o código existente no arquivo Código.gs e cole o código abaixo:

// 1. RECEBER REGISTROS DE TOKENS DO PWA
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    if (data.action === "registerToken") {
      registerToken(data.token, data.userAgent);
      return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "Token registrado!" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
        .setMimeType(ContentService.MimeType.JSON);
  }
}

function registerToken(token, userAgent) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Tokens");
  if (!sheet) {
    sheet = ss.insertSheet("Tokens");
    sheet.appendRow(["Token", "Data de Registro", "Dispositivo"]);
    sheet.getRange("A1:C1").setFontWeight("bold").setBackground("#FBCA03");
  }
  
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === token) {
      sheet.getRange(i + 1, 2).setValue(new Date());
      sheet.getRange(i + 1, 3).setValue(userAgent || "");
      return;
    }
  }
  sheet.appendRow([token, new Date(), userAgent || ""]);
}

// 3. MENU DE NOTIFICAÇÃO NA PLANILHA
function onOpen() {
  var ui = SpreadsheetApp.getUi();
  ui.createMenu('COGERH Notificações')
    .addItem('Disparar Notificação Geral', 'solicitarEnvioNotificacao')
    .addToUi();
}

function solicitarEnvioNotificacao() {
  var ui = SpreadsheetApp.getUi();
  
  if (SERVICE_ACCOUNT.private_key.indexOf("COLE_SUA") !== -1) {
    ui.alert("Erro: Antes de disparar, você precisa colar as credenciais da sua Conta de Serviço do Firebase (geradas nas Configurações do Projeto -> Contas de Serviço) na variável 'SERVICE_ACCOUNT' do script.");
    return;
  }

  var titleResponse = ui.prompt('Título da Notificação', 'Insira o título do aviso:', ui.ButtonSet.OK_CANCEL);
  if (titleResponse.getSelectedButton() !== ui.Button.OK) return;
  var title = titleResponse.getResponseText().trim();
  
  var bodyResponse = ui.prompt('Texto da Notificação', 'Insira o texto da mensagem:', ui.ButtonSet.OK_CANCEL);
  if (bodyResponse.getSelectedButton() !== ui.Button.OK) return;
  var body = bodyResponse.getResponseText().trim();
  
  if (!title || !body) {
    ui.alert("O título e corpo não podem ser vazios.");
    return;
  }
  
  var status = dispararFCM(title, body);
  ui.alert("Resultado do Envio:\n" + status);
}

// Envia a mensagem a todos os tokens cadastrados usando a API FCM V1 (HTTP v1)
function dispararFCM(title, body) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Tokens");
  if (!sheet) return "Nenhum token cadastrado (aba 'Tokens' não existe ainda).";
  
  var data = sheet.getDataRange().getValues();
  var tokens = [];
  for (var i = 1; i < data.length; i++) {
    if (data[i][0]) tokens.push(data[i][0]);
  }
  
  if (tokens.length === 0) return "Nenhum token ativo encontrado na aba 'Tokens'.";
  
  var success = 0;
  var errors = 0;
  var lastError = "";
  
  try {
    var accessToken = getAccessToken_(SERVICE_ACCOUNT);
    var url = "https://fcm.googleapis.com/v1/projects/" + SERVICE_ACCOUNT.project_id + "/messages:send";
    
    for (var i = 0; i < tokens.length; i++) {
      var payload = {
        "message": {
          "token": tokens[i],
          "notification": {
            "title": title,
            "body": body
          },
          "webpush": {
            "notification": {
              "title": title,
              "body": body,
              "icon": "https://raw.githubusercontent.com/diegomaronese/cogerh-app/main/pwa/icon.svg"
            },
            "fcm_options": {
              "link": "https://cogerh-ap.web.app/pwa/"
            }
          }
        }
      };
      
      var options = {
        "method": "post",
        "contentType": "application/json",
        "headers": {
          "Authorization": "Bearer " + accessToken
        },
        "payload": JSON.stringify(payload),
        "muteHttpExceptions": true
      };
      
      try {
        var res = UrlFetchApp.fetch(url, options);
        var code = res.getResponseCode();
        if (code === 200 || code === 201) {
          success++;
        } else {
          errors++;
          lastError = res.getContentText();
        }
      } catch (e) {
        errors++;
        lastError = e.toString();
      }
    }
    
    var resultText = "Enviado via FCM HTTP v1.\n- Sucedidos: " + success + "\n- Falhas: " + errors;
    if (errors > 0) {
      resultText += "\nÚltimo erro recebido: " + lastError;
    }
    return resultText;
  } catch (authErr) {
    return "Erro de Autenticação do Firebase: " + authErr.message;
  }
}

// Gera o token de acesso OAuth2 usando a Conta de Serviço (RS256)
function getAccessToken_(serviceAccount) {
  var header = JSON.stringify({
    "alg": "RS256",
    "typ": "JWT"
  });
  
  var now = Math.floor(Date.now() / 1000);
  var payload = JSON.stringify({
    "iss": serviceAccount.client_email,
    "scope": "https://www.googleapis.com/auth/firebase.messaging",
    "aud": "https://oauth2.googleapis.com/token",
    "exp": now + 3600,
    "iat": now
  });
  
  var toSign = base64EncodeSafe_(header) + "." + base64EncodeSafe_(payload);
  var signatureBytes = Utilities.computeRsaSha256Signature(toSign, serviceAccount.private_key);
  var signature = base64EncodeSafeBytes_(signatureBytes);
  var jwt = toSign + "." + signature;
  
  var options = {
    "method": "post",
    "payload": {
      "grant_type": "urn:ietf:params:oauth:grant-type:jwt-bearer",
      "assertion": jwt
    },
    "muteHttpExceptions": true
  };
  
  var response = UrlFetchApp.fetch("https://oauth2.googleapis.com/token", options);
  var data = JSON.parse(response.getContentText());
  if (data.error) {
    throw new Error("Falha no token OAuth2: " + (data.error_description || data.error));
  }
  return data.access_token;
}

function base64EncodeSafe_(string) {
  var encoded = Utilities.base64EncodeWebSafe(string);
  return encoded.replace(/=+$/, "");
}

function base64EncodeSafeBytes_(bytes) {
  var encoded = Utilities.base64EncodeWebSafe(bytes);
  return encoded.replace(/=+$/, "");
}

Passo 3: Salvar e Implantar como Web App (CRÍTICO)

1. Clique no ícone do Disquete para salvar o projeto.
2. Clique no botão azul Implantar -> Nova implantação.
3. Clique na engrenagem ao lado de "Selecione o tipo" e escolha App da Web.
4. Em "Executar como", selecione Eu (seu e-mail).
5. Em "Quem tem acesso", selecione obrigatoriamente Qualquer pessoa.
6. Clique em Implantar e autorize os acessos da sua conta Google.
7. Copie a URL do aplicativo da Web gerada (ela termina em /exec) e cole-a no campo de configurações do PWA.

⚠️ Ativação de Chaves no Firebase:

No Console Firebase, vá em Configurações do Projeto -> Cloud Messaging. Em "Web push credentials", clique em "Generate key pair" para obter a chave VAPID. Copie o valor e cole-o no campo Chave VAPID no PWA.
Para o disparo de notificações funcionar, gere uma Conta de Serviço em Configurações do Projeto -> Contas de Serviço, clique em Gerar nova chave privada, abra o arquivo JSON baixado e substitua o objeto SERVICE_ACCOUNT do script pelo conteúdo copiado.
